### 2. app.py — 主入口

```python
from flask import Flask
from flask_cors import CORS
from flask_jwt_extended import JWTManager
from config import Config
import sqlite3
import os

def create_app():
    app = Flask(__name__)
    app.config.from_object(Config)

    # 创建上传目录
    os.makedirs(app.config['UPLOAD_FOLDER'], exist_ok=True)

    # CORS - 允许前端跨域访问
    CORS(app, origins=['http://localhost:3000', 'http://localhost:5173'])

    # JWT
    JWTManager(app)

    # 初始化数据库
    with app.app_context():
        init_db()

    # 注册蓝图
    from auth import auth_bp
    from api.user import user_bp
    from api.articles import articles_bp
    from api.health import health_bp
    from api.meals import meals_bp
    from api.posts import posts_bp
    from api.sports import sports_bp
    from api.resources import resources_bp
    from api.mood import mood_bp

    app.register_blueprint(auth_bp, url_prefix='/api/v1/auth')
    app.register_blueprint(user_bp, url_prefix='/api/v1/user')
    app.register_blueprint(articles_bp, url_prefix='/api/v1/articles')
    app.register_blueprint(health_bp, url_prefix='/api/v1/health')
    app.register_blueprint(meals_bp, url_prefix='/api/v1/meals')
    app.register_blueprint(posts_bp, url_prefix='/api/v1/posts')
    app.register_blueprint(sports_bp, url_prefix='/api/v1/sports')
    app.register_blueprint(resources_bp, url_prefix='/api/v1/resources')
    app.register_blueprint(mood_bp, url_prefix='/api/v1/mood')

    return app

def get_db():
    """获取数据库连接"""
    db = sqlite3.connect(Config.DATABASE)
    db.row_factory = sqlite3.Row  # 让查询结果可以用列名访问
    db.execute("PRAGMA foreign_keys = ON")
    return db

def init_db():
    """初始化数据库表"""
    db = get_db()
    with open('schema.sql', 'r', encoding='utf-8') as f:
        db.executescript(f.read())
    db.commit()
    db.close()

if __name__ == '__main__':
    app = create_app()
    app.run(debug=True, port=5000)
```

### 3. utils.py — 工具函数

```python
from flask import jsonify
from flask_jwt_extended import get_jwt_identity
import sqlite3
from config import Config
from datetime import datetime, date

def get_db():
    db = sqlite3.connect(Config.DATABASE)
    db.row_factory = sqlite3.Row
    db.execute("PRAGMA foreign_keys = ON")
    return db

def success(data=None, message='操作成功', code=200):
    """成功响应"""
    resp = {'code': code, 'message': message}
    if data is not None:
        resp['data'] = data
    return jsonify(resp), code

def created(data=None, message='创建成功'):
    """201 响应"""
    return success(data, message, 201)

def error(message='请求失败', code=400, errors=None):
    """错误响应"""
    resp = {'code': code, 'message': message}
    if errors:
        resp['errors'] = errors
    return jsonify(resp), code

def paginated_response(items, total, page, limit):
    """分页响应"""
    return success({
        'items': items,
        'total': total,
        'page': page,
        'limit': limit
    })

def get_current_user_id():
    """从JWT获取当前用户ID"""
    return get_jwt_identity()

def row_to_dict(row):
    """将sqlite3.Row转为字典"""
    if row is None:
        return None
    return dict(row)

def rows_to_list(rows):
    """将sqlite3.Row列表转为字典列表"""
    return [dict(row) for row in rows]

def now():
    """获取当前时间的ISO格式字符串"""
    return datetime.now().strftime('%Y-%m-%d %H:%M:%S')

def today():
    """获取今天的日期字符串"""
    return date.today().strftime('%Y-%m-%d')
```

### 4. auth.py — 认证蓝图

```python
from flask import Blueprint, request
from flask_jwt_extended import create_access_token, jwt_required, get_jwt_identity
from werkzeug.security import generate_password_hash, check_password_hash
from utils import get_db, success, created, error, row_to_dict, now
import re

auth_bp = Blueprint('auth', __name__)

@auth_bp.route('/login', methods=['POST'])
def login():
    data = request.get_json()
    student_id = data.get('studentId', '').strip()
    password = data.get('password', '').strip()

    if not student_id or not password:
        return error('请填写完整信息', 400)

    db = get_db()
    user = db.execute('SELECT * FROM users WHERE student_id = ?', (student_id,)).fetchone()

    if not user or not check_password_hash(user['password'], password):
        return error('学号或密码错误', 401)

    user_dict = row_to_dict(user)
    user_dict.pop('password')  # 不返回密码
    token = create_access_token(identity=str(user['id']))

    return success({
        'user': user_dict,
        'token': token
    }, '登录成功')


@auth_bp.route('/register', methods=['POST'])
def register():
    data = request.get_json()
    student_id = data.get('studentId', '').strip()
    username = data.get('username', '').strip()
    phone = data.get('phone', '').strip()
    password = data.get('password', '').strip()

    # 校验
    if not all([student_id, username, phone, password]):
        return error('请填写完整信息', 400)

    if len(password) < 6:
        return error('密码长度不能少于6位', 400)

    if not re.match(r'^1\d{10}$', phone):
        return error('手机号格式不正确', 400)

    db = get_db()

    # 检查是否重复
    existing = db.execute(
        'SELECT id FROM users WHERE student_id = ? OR username = ?',
        (student_id, username)
    ).fetchone()

    if existing:
        return error('该学号或用户名已被注册', 409)

    # 加密密码
    hashed = generate_password_hash(password)

    # 手机号脱敏存储(同时存一份脱敏的和一份原始的)
    masked_phone = phone[:3] + '****' + phone[7:]

    cursor = db.execute(
        '''INSERT INTO users (student_id, username, password, phone, created_at, updated_at)
           VALUES (?, ?, ?, ?, ?, ?)''',
        (student_id, username, hashed, masked_phone, now(), now())
    )
    db.commit()

    user = db.execute('SELECT * FROM users WHERE id = ?', (cursor.lastrowid,)).fetchone()
    user_dict = row_to_dict(user)
    user_dict.pop('password')

    token = create_access_token(identity=str(user['id']))

    return created({
        'user': user_dict,
        'token': token
    }, '注册成功')


@auth_bp.route('/logout', methods=['POST'])
@jwt_required()
def logout():
    # JWT本身是无状态的，登出主要是前端清除token
    # 如需服务端失效，可在此实现token黑名单
    return success(message='登出成功')
```

### 5. api/user.py — 用户蓝图(示例)

```python
from flask import Blueprint, request
from flask_jwt_extended import jwt_required, get_jwt_identity
from utils import get_db, success, error, row_to_dict, now
from config import Config
import os
import uuid
from werkzeug.utils import secure_filename

user_bp = Blueprint('user', __name__)

@user_bp.route('/profile', methods=['GET'])
@jwt_required()
def get_profile():
    user_id = get_jwt_identity()
    db = get_db()
    user = db.execute('SELECT * FROM users WHERE id = ?', (user_id,)).fetchone()

    if not user:
        return error('用户不存在', 404)

    user_dict = row_to_dict(user)
    user_dict.pop('password')
    return success({'user': user_dict})


@user_bp.route('/profile/basic', methods=['PUT'])
@jwt_required()
def update_basic():
    user_id = get_jwt_identity()
    data = request.get_json()

    db = get_db()

    fields = {}
    if 'username' in data:
        fields['username'] = data['username']
    if 'gender' in data and data['gender'] in [0, 1, 2]:
        fields['gender'] = data['gender']
    if 'height' in data:
        fields['height'] = data['height']
    if 'weight' in data:
        fields['weight'] = data['weight']

    if not fields:
        return error('没有需要更新的字段', 400)

    set_clause = ', '.join(f'{k} = ?' for k in fields.keys())
    values = list(fields.values()) + [now(), user_id]

    db.execute(
        f'UPDATE users SET {set_clause}, updated_at = ? WHERE id = ?',
        values
    )
    db.commit()

    user = db.execute('SELECT * FROM users WHERE id = ?', (user_id,)).fetchone()
    user_dict = row_to_dict(user)
    user_dict.pop('password')

    return success({'user': user_dict}, '基本信息已更新')


@user_bp.route('/profile/contact', methods=['PUT'])
@jwt_required()
def update_contact():
    user_id = get_jwt_identity()
    data = request.get_json()

    fields = {}
    if 'email' in data:
        fields['email'] = data['email']
    if 'phone' in data:
        # 简单手机号脱敏
        phone = data['phone']
        fields['phone'] = phone[:3] + '****' + phone[7:] if len(phone) >= 11 else phone

    if not fields:
        return error('没有需要更新的字段', 400)

    set_clause = ', '.join(f'{k} = ?' for k in fields.keys())
    values = list(fields.values()) + [now(), user_id]

    db = get_db()
    db.execute(f'UPDATE users SET {set_clause}, updated_at = ? WHERE id = ?', values)
    db.commit()

    return success(message='联系方式已更新')


@user_bp.route('/avatar', methods=['POST'])
@jwt_required()
def upload_avatar():
    user_id = get_jwt_identity()

    if 'file' not in request.files:
        return error('请选择文件', 400)

    file = request.files['file']
    if file.filename == '':
        return error('请选择文件', 400)

    # 检查文件类型
    ext = file.filename.rsplit('.', 1)[-1].lower()
    if ext not in Config.ALLOWED_EXTENSIONS:
        return error('仅支持图片格式: png, jpg, jpeg, gif, webp', 400)

    # 生成唯一文件名
    filename = f'{user_id}_{uuid.uuid4().hex[:8]}.{ext}'
    filepath = os.path.join(Config.UPLOAD_FOLDER, filename)
    file.save(filepath)

    avatar_url = f'/uploads/{filename}'

    db = get_db()
    db.execute(
        'UPDATE users SET avatar_url = ?, updated_at = ? WHERE id = ?',
        (avatar_url, now(), user_id)
    )
    db.commit()

    return success({'avatar_url': avatar_url}, '头像上传成功')
```

```javascript
import axios from 'axios'

const request = axios.create({
  baseURL: 'http://localhost:5000/api/v1',
  timeout: 10000
})

// 请求拦截器 - 自动携带token
request.interceptors.request.use(config => {
  const token = sessionStorage.getItem('token') || localStorage.getItem('unihealth_token')
  if (token) {
    config.headers.Authorization = `Bearer ${token}`
  }
  return config
})

// 响应拦截器 - 统一错误处理
request.interceptors.response.use(
  response => response.data,
  error => {
    if (error.response?.status === 401) {
      // token过期，跳转登录
      sessionStorage.clear()
      localStorage.removeItem('userStatus')
      window.location.href = '/login'
    }
    return Promise.reject(error)
  }
)

export default request
```

## 常见问题

### Q: SQLite 支持并发吗？
SQLite 支持读取并发，但写入需要串行。对于校园项目规模的用户量完全够用。生产环境可切换为 MySQL/PostgreSQL。

### Q: 如何处理日期？
SQLite 没有原生 DATE 类型，使用 TEXT 存储 `YYYY-MM-DD` 格式。Python 中用 `date.today().strftime('%Y-%m-%d')`。

### Q: 文件上传的图片如何访问？
需要配置 Flask 的静态文件服务:
```python
from flask import send_from_directory

@app.route('/uploads/<filename>')
def uploaded_file(filename):
    return send_from_directory(app.config['UPLOAD_FOLDER'], filename)
```

### Q: 跨域问题？
已在 app.py 中配置 `flask-cors`，允许 `localhost:3000`（Vite开发服务器）的请求。如果前端端口不同，修改 `origins` 参数。

### Q: 密码安全？
使用 `werkzeug.security.generate_password_hash()` 进行 BCrypt 加密。绝不存储明文密码。

### Q: 后端部署？
1. 使用 `gunicorn` (Linux) 或 `waitress` (Windows) 替代 Flask 内置服务器
2. 生产环境关闭 `debug=True`
3. 更换 `JWT_SECRET_KEY` 为强随机字符串
