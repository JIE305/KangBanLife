## 对接概览

### 当前状态 vs 目标状态

| 方面 | 当前(模拟) | 目标(真实API) |
|------|-----------|-------------|
| 登录 | setTimeout 1.5s 延迟 | POST /auth/login |
| 注册 | setTimeout 1.5s 延迟 | POST /auth/register |
| 文章数据 | 组件内硬编码 | GET /articles/featured |
| 用户资料 | reactive本地对象 | GET/PUT /user/profile |
| 健康数据 | 组件内ref | GET/POST /health/* |
| 匿名树洞 | 组件内ref | GET/POST /posts |
| 配餐/运动/资源 | 组件内硬编码 | GET /meals, /sports/*, /resources/* |

### 对接策略

建议**渐进式对接**，按以下顺序逐个模块迁移:
1. 先对接认证模块（登录/注册）— 这是所有功能的前提
2. 再对接公开数据（文章/配餐/资源）— 不依赖登录状态
3. 最后对接需认证的数据（健康/资料/树洞发帖）

## Step 1: 创建API请求模块

### 创建 `src/utils/request.js`

```javascript
import axios from 'axios'

const request = axios.create({
  baseURL: 'http://localhost:5000/api/v1',
  timeout: 10000,
  headers: {
    'Content-Type': 'application/json'
  }
})

// 请求拦截器：自动附加 token
request.interceptors.request.use(
  config => {
    // 优先从 sessionStorage 取，其次从 localStorage 取
    const token = sessionStorage.getItem('unihealth_token') ||
                  localStorage.getItem('unihealth_token')
    if (token) {
      config.headers.Authorization = `Bearer ${token}`
    }
    return config
  },
  error => Promise.reject(error)
)

// 响应拦截器：统一处理错误
request.interceptors.response.use(
  response => {
    const res = response.data
    // 直接返回 data 字段方便使用
    return res
  },
  error => {
    if (error.response) {
      const { status, data } = error.response

      if (status === 401) {
        // token 过期或无效，清除登录状态
        sessionStorage.removeItem('isLoggedIn')
        sessionStorage.removeItem('user')
        localStorage.removeItem('userStatus')
        localStorage.removeItem('unihealth_token')
        localStorage.removeItem('unihealth_user')
        // 跳转到登录页
        window.location.href = '/login'
      }

      return Promise.reject(data || { message: '网络错误' })
    }
    return Promise.reject({ message: '网络连接失败，请检查后端服务是否启动' })
  }
)

export default request
```

## Step 2: 修改 Pinia Store

### 修改 `src/stores/user.js`

核心改动: 将模拟的 `setTimeout` 替换为真实 API 调用。

```javascript
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { login as loginAPI, register as registerAPI, logout as logoutAPI } from '@/api/auth'
import { getProfile } from '@/api/user'

export const useUserStore = defineStore('user', () => {
  const user = ref(null)
  const isLoggedIn = ref(false)
  const token = ref(null)
  const rememberMe = ref(false)

  const currentUser = computed(() => user.value)

  const login = async (credentials) => {
    const { studentId, password } = credentials

    try {
      const res = await loginAPI({ studentId, password })

      user.value = res.data.user
      token.value = res.data.token
      isLoggedIn.value = true

      // 同步到 sessionStorage (路由守卫用)
      sessionStorage.setItem('isLoggedIn', 'true')
      sessionStorage.setItem('user', JSON.stringify(res.data.user))
      sessionStorage.setItem('unihealth_token', res.data.token)

      // 同步到 localStorage (Navbar 组件用)
      localStorage.setItem('userStatus', 'loggedIn')

      if (rememberMe.value) {
        localStorage.setItem('unihealth_token', res.data.token)
        localStorage.setItem('unihealth_user', JSON.stringify(res.data.user))
      }

      return { success: true, message: res.message || '登录成功' }
    } catch (error) {
      return { success: false, message: error.message || '登录失败' }
    }
  }

  const register = async (userInfo) => {
    const { studentId, username, phone, password } = userInfo

    try {
      const res = await registerAPI({ studentId, username, phone, password })

      user.value = res.data.user
      token.value = res.data.token
      isLoggedIn.value = true

      sessionStorage.setItem('isLoggedIn', 'true')
      sessionStorage.setItem('user', JSON.stringify(res.data.user))
      sessionStorage.setItem('unihealth_token', res.data.token)

      localStorage.setItem('userStatus', 'loggedIn')

      return { success: true, message: res.message || '注册成功' }
    } catch (error) {
      return { success: false, message: error.message || '注册失败' }
    }
  }

  const logout = () => {
    // 调用后端登出(可选)
    logoutAPI().catch(() => {}) // 忽略网络错误

    user.value = null
    token.value = null
    isLoggedIn.value = false
    rememberMe.value = false

    sessionStorage.removeItem('isLoggedIn')
    sessionStorage.removeItem('user')
    sessionStorage.removeItem('unihealth_token')

    localStorage.removeItem('userStatus')
    localStorage.removeItem('unihealth_token')
    localStorage.removeItem('unihealth_user')
  }

  const setRememberMe = (value) => {
    rememberMe.value = value
  }

  const checkAuth = () => {
    const savedToken = localStorage.getItem('unihealth_token')
    const savedUser = localStorage.getItem('unihealth_user')

    if (savedToken && savedUser) {
      try {
        token.value = savedToken
        user.value = JSON.parse(savedUser)
        isLoggedIn.value = true
        rememberMe.value = true

        sessionStorage.setItem('isLoggedIn', 'true')
        sessionStorage.setItem('user', savedUser)
        sessionStorage.setItem('unihealth_token', savedToken)

        localStorage.setItem('userStatus', 'loggedIn')
        return true
      } catch (e) {
        return false
      }
    }
    return false
  }

  // 新增: 从服务器获取最新的用户信息
  const fetchProfile = async () => {
    try {
      const res = await getProfile()
      user.value = res.data.user

      // 更新本地存储
      sessionStorage.setItem('user', JSON.stringify(res.data.user))
      if (rememberMe.value) {
        localStorage.setItem('unihealth_user', JSON.stringify(res.data.user))
      }
    } catch (error) {
      console.error('获取用户信息失败:', error)
    }
  }

  return {
    user, isLoggedIn, token, rememberMe, currentUser,
    login, register, logout, setRememberMe, checkAuth, fetchProfile
  }
})
```

### 修改 `src/stores/health.js`

核心改动: 将本地 ref 数据改为从 API 获取。

```javascript
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import {
  getChecklist, toggleChecklist as toggleAPI,
  getWater, addWater as addWaterAPI,
  getTrend, updateTrend
} from '@/api/health'

export const useHealthStore = defineStore('health', () => {
  const waterAmount = ref(0)
  const dailyGoal = ref(1500)
  const checklist = ref([])
  const trendData = ref([])
  const loading = ref(false)

  const waterProgress = computed(() => {
    return Math.min(100, (waterAmount.value / dailyGoal.value) * 100)
  })

  const completedChecklist = computed(() => {
    return checklist.value.filter(item => item.checked).length
  })

  // 从服务器加载数据
  const fetchChecklist = async (date) => {
    try {
      const res = await getChecklist(date)
      checklist.value = res.data.checklist
    } catch (e) {
      console.error('获取清单失败:', e)
    }
  }

  const toggleChecklistItem = async (itemId) => {
    const item = checklist.value.find(i => i.id === itemId)
    if (!item) return

    const newChecked = !item.checked
    try {
      await toggleAPI(itemId, newChecked)
      item.checked = newChecked
    } catch (e) {
      console.error('更新清单失败:', e)
    }
  }

  const fetchWater = async (date) => {
    try {
      const res = await getWater(date)
      waterAmount.value = res.data.water_ml
    } catch (e) {
      console.error('获取饮水数据失败:', e)
    }
  }

  const addWater = async (amount) => {
    try {
      const res = await addWaterAPI(amount)
      waterAmount.value = res.data.water_ml
    } catch (e) {
      console.error('添加饮水失败:', e)
    }
  }

  const fetchTrend = async (days = 7) => {
    try {
      const res = await getTrend(days)
      trendData.value = res.data.trend
    } catch (e) {
      console.error('获取趋势失败:', e)
    }
  }

  // BMI 计算仍然是纯前端逻辑，不需要API
  const calculateBMI = (height, weight) => {
    if (!height || !weight) return { result: null, status: '' }
    const h = height / 100
    const result = (weight / (h * h)).toFixed(1)
    let status = ''
    if (result < 18.5) status = '偏瘦，建议增加营养'
    else if (result < 24) status = '非常标准'
    else if (result < 28) status = '偏重，建议适当运动'
    else status = '肥胖，建议咨询医生'
    return { result, status }
  }

  return {
    waterAmount, dailyGoal, checklist, trendData, loading,
    waterProgress, completedChecklist,
    addWater, toggleChecklistItem, calculateBMI,
    fetchChecklist, fetchWater, fetchTrend
  }
})
```
